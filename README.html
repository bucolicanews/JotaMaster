Modelagem de Dados & RLS

O JOTA Master utiliza o Supabase (PostgreSQL) com uma arquitetura Multi-tenant. Abaixo está o script completo para criar as tabelas de perfis, módulos e o ecossistema de IA, incluindo as políticas de segurança Row Level Security (RLS).

Script de Instalação (SQL) v1.2 - Resiliente 
-- 1. Criar Tabelas Core
CREATE TABLE IF NOT EXISTS public.profiles (
    id UUID REFERENCES auth.users(id) PRIMARY KEY,
    first_name TEXT,
    role TEXT DEFAULT 'empresa',
    status TEXT DEFAULT 'active'
);

CREATE TABLE IF NOT EXISTS public.system_modules (
    id TEXT PRIMARY KEY,
    name TEXT NOT NULL,
    module_type TEXT DEFAULT 'internal',
    bundle_url TEXT
);

CREATE TABLE IF NOT EXISTS public.ai_skills (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    user_id UUID REFERENCES auth.users(id),
    name TEXT NOT NULL,
    js_code TEXT,
    is_global BOOLEAN DEFAULT false
);

-- 2. Habilitar Segurança (RLS)
ALTER TABLE public.profiles ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.ai_skills ENABLE ROW LEVEL SECURITY;

-- 3. Criar Políticas de Isolamento
CREATE POLICY "Dono vê seus dados" ON public.ai_skills 
FOR SELECT USING (auth.uid() = user_id OR is_global = true);

-- 4. Trigger de Novo Usuário
CREATE OR REPLACE FUNCTION public.handle_new_user() 
RETURNS TRIGGER AS $$
BEGIN
  INSERT INTO public.profiles (id, first_name)
  VALUES (new.id, new.raw_user_meta_data->>'first_name');
  RETURN new;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

* Copie o código acima e cole no SQL Editor do seu painel Supabase.
